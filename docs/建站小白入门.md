### 1.4 路由转发

``` js
export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);

    // 配置：子域名 -> 对应项目域名
    const routeMap = {
      www: "lonemonk.xyz",       // www 转发到根域名主站
      gift: "gift.lonemonk.xyz"  // gift 转发到礼物项目
    };

    // 提取子域名
    const parts = url.host.split(".");
    const sub = parts.length >= 3 ? parts[0] : "";

    // 根域名不经过Worker，直接放行
    if (!sub) {
      return fetch(request);
    }

    // 匹配不到的子域名，默认转发到主站
    const targetHost = routeMap[sub] || routeMap.www;

    const newHeaders = new Headers(request.headers);
    newHeaders.set("Host", targetHost);
    newHeaders.set("X-Forwarded-Proto","https");

    const targetUrl = `https://${targetHost}${url.pathname}${url.search}`;

    return fetch(targetUrl, {
      method: request.method,
      headers: newHeaders,
      body: request.body,
      redirect: "follow"
    });
  }
};
```

``` js
export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);

    // 静态资源后缀匹配
    const staticExtensions = /\.(jpg|jpeg|png|gif|svg|ico|webp|css|js|woff|woff2)$/i;

    // 子域名路由映射
    const routeMap = {
      www: "lonemonk.xyz",
      gift: "gift.lonemonk.xyz"
    };

    // 提取子域名
    const parts = url.host.split(".");
    const sub = parts.length >= 3 ? parts[0] : "";

    // 根域名直接放行
    if (!sub) {
      return fetch(request);
    }

    const targetHost = routeMap[sub] || routeMap.www;
    const newHeaders = new Headers(request.headers);
    newHeaders.set("Host", targetHost);
    newHeaders.set("X-Forwarded-Proto", "https");

    const targetUrl = `https://${targetHost}${url.pathname}${url.search}`;
    
    // 构建请求
    const fetchRequest = new Request(targetUrl, {
      method: request.method,
      headers: newHeaders,
      body: request.body,
      redirect: "follow"
    });

    // 静态资源开启 Cloudflare 边缘缓存
    if (staticExtensions.test(url.pathname)) {
      // 强制让 CF 缓存该请求
      ctx.passiveOnFetchEvent(true);
      fetchRequest.headers.set('Cache-Control', 'public, max-age=31536000, s-maxage=31536000, immutable');
    }

    // 发起请求
    let response = await fetch(fetchRequest);

    // 静态资源：自定义缓存头，优化浏览器缓存
    if (staticExtensions.test(url.pathname)) {
      response = new Response(response.body, response);
      // 浏览器缓存 1 年 + Cloudflare 边缘永久缓存 + 不可变资源
      response.headers.set('Cache-Control', 'public, max-age=31536000, s-maxage=31536000, immutable');
      // 启用 CF 边缘缓存
      response.headers.set('CDN-Cache-Control', 'public, max-age=31536000');
      // 解决跨域图片/字体问题
      response.headers.set('Access-Control-Allow-Origin', '*');
    }

    return response;
  }
};
```
